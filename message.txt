def giveSquad(data):
    n = len(data)
    
    dp = {}
    teamSquad = []
    
    def f(idx,budget,maxGoalKeeper,maxDefender,maxMidFielder,maxForward):        
        
        if(maxGoalKeeper == 0 and maxDefender == 0 and maxMidFielder == 0 and maxForward == 0):
            return 0
        
        if(idx == n):
            if(maxGoalKeeper != 0 or maxDefender != 0 or maxMidFielder != 0 or maxForward != 0): 
                return -math.inf
            return 0
        
        if((idx,budget,maxGoalKeeper,maxDefender,maxMidFielder,maxForward,tuple(takenTeam)) in dp):
            return dp[(idx,budget,maxGoalKeeper,maxDefender,maxMidFielder,maxForward,tuple(takenTeam))]
        
        
        name,pos,team,id,val,wt = data[idx]
        pick = -math.inf
        if(budget >= wt and takenTeam[teamsDic[team]] < teamLimit and val != 0):
            if(pos == "DEF"):
                if(maxDefender > 0):
                    takenTeam[teamsDic[team]] += 1
                    pick = val + f(idx + 1,budget - wt,maxGoalKeeper,maxDefender - 1,maxMidFielder,maxForward)
                    takenTeam[teamsDic[team]] -= 1
            elif(pos == "MID"):
                if(maxMidFielder > 0):
                    takenTeam[teamsDic[team]] += 1
                    pick = val + f(idx + 1,budget - wt,maxGoalKeeper,maxDefender,maxMidFielder - 1,maxForward)
                    takenTeam[teamsDic[team]] -= 1
            elif(pos == "FWD"):
                if(maxForward > 0):
                    takenTeam[teamsDic[team]] += 1
                    pick = val + f(idx + 1,budget - wt,maxGoalKeeper,maxDefender,maxMidFielder,maxForward - 1)
                    takenTeam[teamsDic[team]] -= 1
            else:
                if(maxGoalKeeper > 0):
                    takenTeam[teamsDic[team]] += 1
                    pick = val + f(idx + 1,budget - wt,maxGoalKeeper - 1,maxDefender,maxMidFielder,maxForward)
                    takenTeam[teamsDic[team]] -= 1
        notPick = f(idx + 1,budget,maxGoalKeeper,maxDefender,maxMidFielder,maxForward)
        
        if(pick > notPick):
            teamSquad.append(idx)
            dp[(idx,budget,maxGoalKeeper,maxDefender,maxMidFielder,maxForward,tuple(takenTeam))] = pick
            return pick
        dp[(idx,budget,maxGoalKeeper,maxDefender,maxMidFielder,maxForward,tuple(takenTeam))] = notPick
        return notPick
        
        
    f(0,budget,1,3,5,3)
    return teamSquad
